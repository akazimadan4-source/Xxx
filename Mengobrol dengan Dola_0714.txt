local Road = W:FindFirstChild("Map") and W.Map:FindFirstChild("Roads/Sidewalks")
local Noc = {}
local function SetCol(p,v) pcall(function() p.CanCollide = v end) end
local function Excl(p) return (Road and p:IsDescendantOf(Road)) or p:IsDescendantOf(LP.Character) or p:IsA("VehicleSeat") end
local function Noclip()
    local r = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
    if not r then return end
    for _,p in W:GetDescendants() do
        if p:IsA("BasePart") and not Excl(p) then
            if not Noc[p] then Noc[p] = { CanCollide = p.CanCollide } SetCol(p,false) end
        end
    end
end
local function ResetNoc() for p,v in pairs(Noc) do if p:IsA("BasePart") then SetCol(p,v.CanCollide) end end Noc = {} end

local LookConn
LookDown = function(Z,F)
    if LookConn then return end
    LookConn = RunS.RenderStepped:Connect(function()
        local r = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
        if not r then return end
        local p = r.Position
        local f = Vector3.new(r.CFrame.LookVector.X, 0, r.CFrame.LookVector.Z).Unit
        local t = CFrame.lookAt(p, p+f)
        if not F then t = t * CFrame.Angles(math.rad(-80), 0, 0) end
        Cam.CFrame = CFrame.lookAt(p - (t.LookVector * Z), p)
    end)
end

Teleport = function(Dest)
    local mt = Cfg.AutoFarm.MovementType
    if mt == "Safe Mode" then
        if LP.Character and LP.Character.Humanoid.SeatPart and LP.Character.Humanoid.SeatPart.Name == "DriveSeat" then Notif("Safe","Unmount bike first",3) return end
        local r = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
        if not r then return "Failed" end
        SM_Root = r
        SM_Conn = RunS.Stepped:Connect(Noclip)
        SafeRun = true
        local u = Vector3.new(r.Position.X, -1, r.Position.Z)
        local tw = TS:Create(r, TweenInfo.new(1, Enum.EasingStyle.Linear), { CFrame = CFrame.new(u) * CFrame.Angles(0, math.atan2(r.CFrame.LookVector.X, r.CFrame.LookVector.Z), 0) })
        tw:Play(); tw.Completed:Wait()
        local g = Vector3.new(Dest.X, -2.137, Dest.Z)
        LookDown(15, false)
        while anyFarm() and r and r.Parent and (Vector3.new(r.Position.X,0,r.Position.Z) - Vector3.new(g.X,0,g.Z)).Magnitude > 1.8 do
            local dir = (g - r.Position).Unit
            r.CFrame = CFrame.new(Vector3.new(r.Position.X + dir.X, -2.137, r.Position.Z + dir.Z), Vector3.new(g.X, -2.137, g.Z))
            task.wait(.05)
        end
        if LookConn then LookConn:Disconnect(); LookConn = nil end
        if not anyFarm() then CleanSafe(); return "Stopped" end
        r.CFrame = Dest
        task.wait(.1)
        CleanSafe()
        return "Success"
    elseif mt == "Bike" then
        local b = getgenv().Find_Bike()
        if not b then Notif("Bike","Not found",3) return "Failed" end
        if LP.Character and LP.Character.Humanoid.SeatPart and LP.Character.Humanoid.SeatPart.Name == "DriveSeat" then
            for _=1,30 do b:SetPrimaryPartCFrame(Dest + Vector3.new(0,1.5,0)) end
            Notif("Bike","Teleported!",3)
        else Notif("Bike","Sit on bike first",3) end
        return "Success"
    end
end

BuyApartment = function()

        local function Own(a) local B=a:FindFirstChild("Board") local L=B and B:FindFirstChild("name") and B.name:FindFirstChild("SurfaceGui") and B.name.SurfaceGui:FindFirstChild("TextLabel") return L and L.Text==LP.Name end

-- Sistem Noclip Aman
local Road = W:FindFirstChild("Map") and W.Map:FindFirstChild("Roads/Sidewalks")
local Noc = {}

local function SetCol(p, v) 
    pcall(function() p.CanCollide = v end) 
end

local function Excl(p) 
    return (Road and p:IsDescendantOf(Road)) 
        or p:IsDescendantOf(LP.Character) 
        or p:IsA("VehicleSeat") 
end

local function Noclip()
    local r = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
    if not r then return end
    for _,p in W:GetDescendants() do
        if p:IsA("BasePart") and not Excl(p) then
            if not Noc[p] then 
                Noc[p] = { CanCollide = p.CanCollide } 
                SetCol(p, false) 
            end
        end
    end
end

local function ResetNoc() 
    for p,v in pairs(Noc) do 
        if p:IsA("BasePart") then SetCol(p, v.CanCollide) end 
    end 
    Noc = {} 
end

Teleport = function(Dest)
    local mt = Cfg.AutoFarm.MovementType
    if mt == "Safe Mode" then
        if LP.Character and LP.Character.Humanoid.SeatPart 
        and LP.Character.Humanoid.SeatPart.Name == "DriveSeat" then 
            Notif("Safe","Unmount bike first",3) 
            return

        SM_Root = r

        local u = Vector3.new(r.Position.X, -1, r.Position.Z)
        local tw = TS:Create(
            r, 
            TweenInfo.new(1, Enum.EasingStyle.Linear), 
            { 
                CFrame = CFrame.new(u) 
                    * CFrame.Angles(0, math.atan2(r.CFrame.LookVector.X, r.CFrame.LookVector.Z), 0) 
            }
        )
        tw:Play(); tw.Completed:Wait()
        
        local g = Vector3.new(Dest.X, -2.137, Dest.Z)
        LookDown(15, false)
        
        while anyFarm() and r and r.Parent 
        and (Vector3.new(r.Position.X, 0, r.Position.Z) - Vector3.new(g.X, 0, g.Z)).Magnitude > 1.8 do
            local dir = (g - r.Position).Unit
            r.CFrame = CFrame.new(
                Vector3.new(r.Position.X + dir.X, -2.137, r.Position.Z + dir.Z), 
                Vector3.new(g.X, -2.137, g.Z)
            )
            task.wait(0.05)
        end
        
        if LookConn then LookConn:Disconnect(); LookConn = nil end
        if not anyFarm() then CleanSafe(); return "Stopped" end
        
        r.CFrame = Dest
        task.wait(0.1)
        CleanSafe()
        return "Success"

    elseif mt == "Bike" then
        local b = getgenv().Find_Bike()
        if not b then Notif("Bike","Not found",3); return "Failed" end
        if LP.Character and LP.Character.Humanoid.SeatPart

        else 
            Notif("Bike","Sit on bike first",3) 
        end
        return "Success"
    end
end

local function chipMoveTo(npc) 
    if not chipActive then return false end 
    local t = npc and npc:FindFirstChild("HumanoidRootPart") and npc.HumanoidRootPart.Position 
    if not t then return false end

end

local function startChipFarm()
    startCamLock()
    task.spawn(function()
        local store = RS:WaitForChild("RemoteEvents"):WaitForChild("StorePurchase")
        local Homeless = RS:FindFirstChild("Workspace") and RS.Workspace:FindFirstChild("Homeless")
        if not Homeless then warn("Chip: Homeless missing"); chipActive=false; hideDet(); stopCamLock(); return end
        local Live = W.Folders.HomelessPeople
        local nums = {"One","Two","Three","Four","Five","Six","Seven","Eight","Nine","Ten","Eleven","Twelve","Thirteen","Fourteen","Fifteen"}

        local function getSit(f) 
            local sit = {} 
            if not f then return sit end 
            for _,npc in f:GetChildren() do 
                for _,name in ipairs(nums) do 
                    if npc.Name == name and npc:GetAttribute("AnimationState") == "Sitting" then 
                        table.insert(sit,npc); break 
                    end 
                end 
            end 
            return sit 
        end

        while chipActive do
            local c = LP.Character; local h = c and c:FindFirstChildOfClass("Humanoid")
            if not c or not h or h.Health <= 0 then waitAlive(); goto cont end

            local targets = getSit(Homeless)
            showDet(); updDet(#targets,false)
            local ws = 0
            while chipActive and #targets == 0 and ws <= 300 do task.wait(1); targets=getSit(Homeless); updDet(#targets,false); ws=ws+1 end
            if not chipActive then hideDet(); break end
            if #targets == 0 then hideDet(); goto cont end

            updDet(#targets,true); task.wait(1); hideDet()

            if chipMove(Vector3.new(-771,4,-198)) == "Stopped" then break end
            local bc = 0
            while chipActive and (count("Potato") < #targets or count("Flour") < #targets) do
                bc = bc+1; if bc > 60 then break end
                if count("Potato") < #targets then pcall(function() store:FireServer("Potato") end); task.wait(0.4) end
                if chipActive and count("Flour") < #targets then pcall(function() store:FireServer("Flour") end); task.wait(0.4) end
            end
            if not chipActive then break end

            local pc = 0
            repeat
                if not chipActive then break end
                pc = pc+1; if pc > 10 then break end

                if chipMove(Vector3.new(-479,4,-436)) == "Stopped" then break end; pressE(); task.wait(1)
                if chipMove(Vector3.new(-463,4,-469)) == "Stopped" then break end; equip("Potato")
                local cut = W.Map.Locations["The Laboratory"]["Cutting Boards"]["Potato Cutter"].Model.Union.Attachment.ProximityPrompt
                if cut then fire(cut) end; task.wait(5.7)

                if not chipActive then break end
                if chipMove(Vector3.new(-462,4,-475)) == "Stopped" then break end
                local bag = W.Map.Locations["The Laboratory"].Prompts["Plastic Bag"].Attachment.ProximityPrompt
                if bag then fire(bag) end; task.wait(5.7)

                if not chipActive then break end
                if chipMove(Vector3.new(-487,4,-536)) == "Stopped" then break end; equip("Flour")
                local bowl = W.Map.Locations["The Laboratory"].Bowls:GetChildren()[2].ProximityPrompt
                if bowl then fire(bowl) end; task.wait(5.7)

                if not chipActive then break end
                if chipMove(Vector3.new(-521,4,-465)) == "Stopped" then break end
                local pot = W.Map.Locations["The Laboratory"].Pots:GetChildren()[7].ProximityPrompt
                if pot then fire(pot) end
                for _=1,65 do if not chipActive then break end task.wait(1) end
                if chipActive then pressE(); task.wait(1) end
            until not chipActive or (count("Potato") == 0 and count("Flour") == 0)

            if not chipActive then break end
            if chipMove(Vector3.new(-33,4,-25)) == "Stopped" then break end

            local poor = W.Folders.NPCs:FindFirstChild("Poor Guy")
            if not poor then task.wait(2); goto cont end
            local pp = poor.UpperTorso:FindFirstChildOfClass("ProximityPrompt")
            if pp then fire(pp); task.wait(1.5) end

            local ds = 0
            while chipActive and count("Hot Chips") > 0 do
                ds = ds+1; if ds>50 then break end
                local all = getSit(Homeless)
                for _,v in getSit(Live) do table.insert(all,v) end
                if #all == 0 then task.wait(1)
                else
                    for _,npc in ipairs(all) do
                        if not chipActive or count("Hot Chips") <= 0 then break end
                        if npc and npc:GetAttribute("AnimationState") == "Sitting" then
                            local c2 = LP.Character; local h2 = c2 and c2:FindFirstChildOfClass("Humanoid")
                            if not c2 or not h2 or h2.Health <= 0 then waitAlive(); break end
                            if chipMoveTo(npc) then
                                equip("Hot Chips")
                                Cfg.AutoFarm.Chips.Status.Earned = Cfg.AutoFarm.Chips.Status.Earned + 300
                                Cfg.AutoFarm.Chips.Status.Delivered = Cfg.AutoFarm.Chips.Status.Delivered + 1
                                task.wait(0.3); pressE(); task.wait(1.5)
                            end
                        end
                    end
                end
                task.wait(0.5)
            end
            ::cont::
        end
        hideDet(); stopCamLock()
    end)
end
